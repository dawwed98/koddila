#Pogrupuj tabel w taki sposb, aby otrzyma redni przychd (revenue) oraz redni budet (budget) w danym roku dla filmw opublikowanych od 2010 (wcznie) do 2016 roku (wcznie). Nastpnie na tej podstawie stwrz wykres, w ktrym rednie przychody s wykresem kolumnowym, a redni budet wykresem liniowym na tych samych osiach. Sformatuj odpowiednio o X oraz o Y. Dodaj tytu wykresu, oraz legend, ktra znajduje si w prawym grnym rogu ptna, lecz poza obszarem osi. Przykad wykresu widoczny poniej:

# Przygotowanie danych do wykresu
df['release_date'] = pd.to_datetime(df['release_date'], 
errors='coerce')

# Filtrowanie danych do lat 2010-2016
df_filtred = df[(df['release_date'].dt.year >= 2010) & (df['release_date'].dt.year <= 2016)]

# Grupowanie danych wedug roku i obliczanie rednich wartoci
grouped_data = df_filtred.groupby(df_filtred['release_date'].dt.year).agg({'revenue': 'mean', 'budget': 'mean'})
grouped_data = grouped_data.rename(columns={'revenue': 'redni przychd', 'budget': 'redni budet'})

# Tworzenie wykresu
fig, ax1 = plt.subplots(figsize=(12, 6))

# Wykres supkowy dla redniego budetu
bar_width = 0.4
ax1.bar(grouped_data.index - bar_width / 2, grouped_data['redni budet'], width=bar_width, label='redni budet', color='blue')
ax1.set_xlabel('Rok')
ax1.set_ylabel('redni budet', color='green')
ax1.tick_params(axis='y', labelcolor='yellow')

# Drugi wykres (lininiarny) dla redniego budetu
ax2 = ax1.twinx()
ax2.plot(grouped_data.index + bar_width / 2, grouped_data['redni przychd'], color='red', marker='o', label='redni przychd')
ax2.set_ylabel('redni przychd', color='pink')
ax2.tick_params(axis='y', labelcolor='purple')

# Ustawienia osi x
ax1.set_xticks(grouped_data.index)
ax1.set_xlim(2009.5, 2016.5)

# Ustawienia osi y
ax1.yaxis.set_major_formatter(plt.FuncFormatter(lambda x, _: f'{x / 1e6:.0f}M'))
ax2.yaxis.set_major_formatter(plt.FuncFormatter(lambda x, _: f'{x / 1e6:.0f}M'))

# Dodanie legendy i tytuu
plt.title('redni budet i przychd filmw w latach 2010-2016')
lines, labels = ax1.get_legend_handles_labels()
lines2, labels2 = ax2.get_legend_handles_labels()
ax2.legend(lines + lines2, labels + labels2, loc='upper left')

# Wyswietlenie wykresu
plt.tight_layout()
plt.grid()
plt.show()

#Filmy, ktrego gatunku trwaj rednio najduej (runtime)?

# czenie baz danych na podstawie 'genre_id' i dodanie kolumny 'runtime'
merged_data_with_runtime = pd.merge(
	baza_filmw[['original_title', 'genre_id', 'runtime']],
	baza_gatunkw.rename(columns={'Unnamed: 0': 'genre_id', 'genres': 'name'}),
	how='inner',
	left_on='genre_id',
	right_on='genre_id'
)

# Obliczanie redniego czasu trwania filmw dla kadego gatunku
# .mean() oblicza redni warto dla kadej grupy
# .groupby() grupuje dane wedug kolumny 'name' (gatunek)
gatunki_time = merged_data_with_runtime['runtime'].groupby(merged_data_with_runtime['name']).mean()
# Sortowanie wynikw malejco i wybieranie 10 najduszych gatunkw
# .sort_values(ascending=False) sortuje dane malejco
gatunki_time = gatunki_time.sort_values(ascending=False).head(10)
# Wywietlanie wynikw .idxmax() zwraca indeks (gatunek) o najwikszej wartoci, a .max() zwraca t warto
print(gatunki_time.idxmax(), gatunki_time.max())
